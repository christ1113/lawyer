<script>
// Navigation scroll effect
const navbar = document.getElementById('navbar');
window.addEventListener('scroll', () => {
    if (window.scrollY > 50) {
        navbar.classList.add('scrolled');
    } else {
        navbar.classList.remove('scrolled');
    }
});

// Mobile menu toggle
const navToggle = document.getElementById('navToggle');
const navMenu = document.getElementById('navMenu');

navToggle.addEventListener('click', () => {
    navMenu.classList.toggle('active');
});

// Close mobile menu when clicking on a link
const navLinks = document.querySelectorAll('.nav-menu a');
navLinks.forEach(link => {
    link.addEventListener('click', () => {
        navMenu.classList.remove('active');
    });
});

// Smooth scroll for navigation links
document.querySelectorAll('a[href^="#"]').forEach(anchor => {
    anchor.addEventListener('click', function (e) {
        const href = this.getAttribute('href');
        if (href !== '#' && href !== '#contact') {
            e.preventDefault();
            const target = document.querySelector(href);
            if (target) {
                const offset = 80;
                const targetPosition = target.offsetTop - offset;
                window.scrollTo({
                    top: targetPosition,
                    behavior: 'smooth'
                });
            }
        }
    });
});

// Fetch case stories from API
// Use the backend service (port 3307) directly so the frontend static server (nginx on 8081)
// can still fetch dynamic data in local development.
async function fetchCaseStories() {
    try {
        const apiUrl = 'http://localhost:3307/backend.php?action=list';
        const response = await fetch(apiUrl, { credentials: 'include' });
        if (!response.ok) {
            throw new Error('Failed to fetch case stories');
        }
        const data = await response.json();
        return data.success ? data.data : [];
    } catch (error) {
        console.error('Error fetching case stories:', error);
        return [];
    }
}

// Render case stories to the page
function renderCaseStories(stories) {
    const storiesGrid = document.querySelector('.stories-grid');
    if (!storiesGrid || !stories.length) return;

    storiesGrid.innerHTML = stories.map(story => `
        <div class="story-card">
            <div class="story-header">
                <span class="story-date">${story.upload_date}</span>
                <span class="story-lawyer">æ¿è¾¦å¾å¸«ï¼${story.lawyer}</span>
            </div>
            <h3 class="story-title">${story.title} - ${story.client_name}</h3>
            <p class="story-content">${story.description}</p>
        </div>
    `).join('');
}

// Initialize case stories
async function initCaseStories() {
    const stories = await fetchCaseStories();
    renderCaseStories(stories);
}

// Load case stories when page loads
document.addEventListener('DOMContentLoaded', initCaseStories);
</script>
